041             * WMap's private class to use compass functions
042             * @author Juan Antonio Jimenez Lopez < ruchin10@gmail.com >
043             *
044             */
045            private class WOverlay extends MyLocationOverlay{
046                    ArrayList<WMapListener> listeners = new ArrayList<WMapListener>();
047                    public WOverlay(Context context, WView WView) {
048                            super(context, WView);
049                            
050                    }
051                    @Override
052                    public boolean onTap(GeoPoint hitpoint, MapView WView)
053                    {
054                            for(int i=0; i<listeners.size(); i++)
055                                    listeners.get(i).onMapTapped(WPoint.toWPoint(hitpoint));
056                            return super.onTap(hitpoint, WView);
057                            //return false;
058                    }
059                    /**
060                     * Sets a new WMapListener to the listeners list
061                     * @param listener
062                     */
063                    public void setWMapListener(WMapListener listener)
064                    {
065                            listeners.add(listener);
066                    }
067                    
068            }

227                    if(Math.abs(center_lat) > 90.0 || Math.abs(center_lon) > 179.0)

237                            }
238                            // center error, zoom ok
239                            else{
240                                    map_controller.setZoom(zoom_p);
241                                    throw new WMapException("Center out of range. Center changed to (" + (center_aux.getLatitudeE6()/1E6) +", " + (center_aux.getLongitudeE6()/1E6) +").");

251                            throw new WMapException("Center and Zoom out of range. Zoom changed to 21.");
