001    package com.iscbd.wingmaps.map;
002    
003    import java.util.ArrayList;
004    
005    import android.content.Context;
006    import android.graphics.drawable.Drawable;
007    import android.util.AttributeSet;
008    import android.view.MotionEvent;
009    import android.view.View;
010    
011    import com.google.android.maps.GeoPoint;
012    import com.google.android.maps.ItemizedOverlay;
013    import com.google.android.maps.MapView;
014    import com.google.android.maps.MyLocationOverlay;
015    import com.google.android.maps.OverlayItem;
016    
017    public class WView extends MapView {
018            
019            MyItemizedOverlay myItemizedOverlay = null;
020             MyLocationOverlay myLocationOverlay = null;
021             
022            /*public WView(Context context, AttributeSet attrs) {
023                    super(context, attrs);
024            }*/
025    
026            @Override
027            public boolean onTouchEvent(MotionEvent ev) {
028                    return super.onTouchEvent(ev);
029            }
030            
031            
032            public WView(Context context, AttributeSet attrs) {
033                      super(context, attrs);
034                      init(context);
035                     }
036                     
037                     public WView(Context context, AttributeSet attrs, int defStyle) {
038                      super(context, attrs, defStyle);
039                      init(context);
040                     }
041                      
042                     private void init(Context ctx){
043                      setClickable(true);
044                      setBuiltInZoomControls(true);
045                       
046                      Drawable marker=getResources().getDrawable(android.R.drawable.star_big_on);
047                      int markerWidth = marker.getIntrinsicWidth();
048                      int markerHeight = marker.getIntrinsicHeight();
049                      marker.setBounds(0, markerHeight, markerWidth, 0);
050                       
051                      myItemizedOverlay = new MyItemizedOverlay(marker);
052                      getOverlays().add(myItemizedOverlay);
053                       
054                      myLocationOverlay = new MyLocationOverlay(ctx, this);
055                      getOverlays().add(myLocationOverlay);
056                       
057                     }
058                      
059                     public void addMarker(GeoPoint p, String title, String snippet){
060                      myItemizedOverlay.addItem(p, title, snippet);
061                     }
062                      
063                     public void callOnResume(){
064                      myLocationOverlay.enableMyLocation();
065                      myLocationOverlay.enableCompass();
066                     }
067                      
068                     public void callOnPause(){
069                      myLocationOverlay.disableMyLocation();
070                      myLocationOverlay.disableCompass();
071                     }
072                      
073                     public class MyItemizedOverlay extends ItemizedOverlay<OverlayItem> {
074                       
075                      private ArrayList<OverlayItem> overlayItemList = new ArrayList<OverlayItem>();
076                     
077                      public MyItemizedOverlay(Drawable defaultMarker) {
078                       //super(defaultMarker);
079                       super(boundCenterBottom(defaultMarker));
080                       populate();
081                      }
082                     
083                      @Override
084                      protected OverlayItem createItem(int i) {
085                       return overlayItemList.get(i);
086                      }
087                     
088                      @Override
089                      public int size() {
090                       return overlayItemList.size();
091                      }
092                       
093                      public void addItem(GeoPoint p, String title, String snippet){
094                       OverlayItem newItem = new OverlayItem(p, title, snippet);
095                       overlayItemList.add(newItem);
096                       populate(); 
097                      }
098                     
099                     }
100            
101    }
