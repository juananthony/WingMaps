001    package com.iscbd.wingmaps.actor;
002    
003    import android.location.Location;
004    import android.location.LocationListener;
005    import android.location.LocationManager;
006    import android.os.Bundle;
007    
008    public class LocationListenerProxy implements LocationListener {
009            private final LocationManager mLocationManager;
010            private LocationListener mListener = null;
011    
012            public LocationListenerProxy(final LocationManager pLocationManager) {
013                    mLocationManager = pLocationManager;
014            }
015    
016            public boolean startListening(final LocationListener pListener, final long pUpdateTime,
017                            final float pUpdateDistance) {
018                    boolean result = false;
019                    mListener = pListener;
020                    for (final String provider : mLocationManager.getProviders(true)) {
021                            if (LocationManager.GPS_PROVIDER.equals(provider)
022                                            || LocationManager.NETWORK_PROVIDER.equals(provider)) {
023                                    result = true;
024                                    mLocationManager.requestLocationUpdates(provider, pUpdateTime, pUpdateDistance,
025                                                    this);
026                            }
027                    }
028                    return result;
029            }
030    
031            public void stopListening() {
032                    mListener = null;
033                    mLocationManager.removeUpdates(this);
034            }
035    
036            @Override
037            public void onLocationChanged(final Location arg0) {
038                    if (mListener != null) {
039                            mListener.onLocationChanged(arg0);
040                    }
041            }
042    
043            @Override
044            public void onProviderDisabled(final String arg0) {
045                    if (mListener != null) {
046                            mListener.onProviderDisabled(arg0);
047                    }
048            }
049    
050            @Override
051            public void onProviderEnabled(final String arg0) {
052                    if (mListener != null) {
053                            mListener.onProviderEnabled(arg0);
054                    }
055            }
056    
057            @Override
058            public void onStatusChanged(final String arg0, final int arg1, final Bundle arg2) {
059                    if (mListener != null) {
060                            mListener.onStatusChanged(arg0, arg1, arg2);
061                    }
062            }
063    }
