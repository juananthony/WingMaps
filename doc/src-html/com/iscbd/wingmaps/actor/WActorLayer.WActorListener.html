099            // Counter to follow and adjust map
100            final CountDownTimer counter = new CountDownTimer(8000 , 1000) { 
101                    public void onTick(long millisUntilFinished) { }
102                    public void onFinish() {
103                            // When the counter finish change the center and zoom
104                            if(mFollow)
105                                    wmap.setCenter(getMainActor().getPosition());
106                            if(mAdjust)
107                            {
108                                    WPoint first_farthest = actors.get(getFarthestActor(FIRST_FARTHEST_ACTOR)).getPosition();
109                                    wmap.adjustMap(first_farthest);
110                                    wmap.invalidate();
111                            }
112                            else
113                                    
114                        counter.start(); 
115                    }
116                    };

141            public WActorLayer(WMap wmap_p, WMarker defaultMarker_p, Context context) {
142                      this.context = context;
143                      //sensorManager = (SensorManager)context.getSystemService(Context.SENSOR_SERVICE);

153                      mAdjust = false;              // Adjust map to actors

155                      
156                      WPoint geoloc = new WPoint(37913350,-4717208);
157                      this.enableMyLocation();
158                      if(mLocation != null)
159                              geoloc = new WPoint((int)(mLocation.getLatitude()*1E6), (int)(mLocation.getLongitude()*1E6));

161                      defaultMarker = defaultMarker_p;
162                      main_actor = new WActor(context, 
163                                            -1, 
164                                            "User", 
165                                            "", 
166                                            geoloc,
167                                            true, 
168                                            defaultMarker);
169                      
170                      actors.add(main_actor);
171                      actors_number++;
172                      actors_ordered = getActorsOrderedByLatitude();
173                      this.disableMyLocation();
174                      
175                      
176            }

245             public boolean enableMyLocation() {
246                     boolean result = true;
247    
248                            if (mLocationListener == null) {
249                                    mLocationListener = new LocationListenerProxy(mLocationManager);
250                                    result = mLocationListener.startListening(this, mLocationUpdateMinTime,
251                                                    mLocationUpdateMinDistance);
252                            }
253                            
254                            mLocation = mLocationManager.getLastKnownLocation(mLocationManager.GPS_PROVIDER);
255                            
256                            // Update the screen to see changes take effect
257                            if (wmap != null) {
258                                    if(mFollow)
259                                    {
260                                            if(mLocation != null)
261                                                    wmap.setCenter(new WPoint((int)(mLocation.getLatitude()*1E6), (int)(mLocation.getLongitude()*1E6)));
262                                    }
263                                    //wmap.postInvalidate();
264                            }
265    
266                            return result;
267             }

291            private WActor getActorTapped(WPoint hitpoint, MapView mapView)
292            {
293                    int actor_index=-1;
294                    WActor actor_tapped = null;
295                    actors_ordered = getActorsOrderedByLatitude();
296                    Point point_hit = new Point();
297                    Projection projection = mapView.getProjection();
298                    projection.toPixels(hitpoint, point_hit);
299                    for(int i=actors_ordered.size()-1; i>=0; i--)
300                    {
301                            if(actors_ordered.get(i).isHit(point_hit.x,point_hit.y,mapView))
302                            {
303                                    actor_tapped = actors_ordered.get(i);
304                                    break;
305                            }
306                    }
307                    return actor_tapped;
308            }

311            @Override
312            public boolean onTap(GeoPoint hitpoint, MapView mapView)
313            {
314                    if(isVisible())
315                            if(isActive())
316                            {
317                                    WActor actor_tapped = getActorTapped(WPoint.toWPoint(hitpoint), mapView);
318                                    if(actor_tapped != null)
319                                    {
320                                            for(int i=0; i < listeners.size(); i++)
321                                                    listeners.get(i).onActorTapped(actor_tapped, actors.indexOf(actor_tapped));
322                                            return true;
323                                    }
324                                    else
325                                            return super.onTap(hitpoint, mapView);
326                            }
327                            else
328                                    return super.onTap(hitpoint, mapView);
329                    else
330                            return super.onTap(hitpoint, mapView);
331            }

756            /**
757             * Follow the main actor.
758             * @param condition If it is true, the map is going to follow the main actor
759             * @param fix If 'fix' is true, is impossible change the center of the map, if it is false, the center change every 8 seconds.
760             */
761            public void setFollowMainActor(boolean condition, boolean fix)
762            {
763                    if(fix)
764                    {
765                            mFollowFix = condition;
766                            mFollow = false;
767                    }
768                    else
769                    {
770                            if(condition)
771                                    wmap.setCenter(getMainActor().getPosition());
772                            mFollow = condition;
773                            mFollowFix = false;
774                    }
775            }

777            /**
778             * 
779             * @param condition
780             * @param fix If it is true
781             */
782            public void setAdjustMapToActors(boolean condition, boolean fix)
783            {
784                    if(fix)
785                    {
786                            mAdjustFix = condition;
787                            mAdjust = false;
788                    }
789                    else
790                    {
791                            WPoint first_farthest = actors.get(getFarthestActor(FIRST_FARTHEST_ACTOR)).getPosition();
792                            if(condition)
793                                    wmap.adjustMap(first_farthest);

800            /**
801             * Get the index of farthest (or second farthest) actor from the main actor.
802             * @param flag That flag allow to get the farthest actor (with 'FIRST_FARTHEST_ACTOR') from main actor or the second farthest actor (with 'SECOND_FARTHEST_ACTOR').
803             * @return The index of the farthest (or second farthest) actor from main actor.
804             */
805            public int getFarthestActor(int flag)
806            {
807                    int return_value = -1;
808                    int first_farthest = 1;
809                    int second_farthest = 2;
810                    double distance;
811                    double farthest_distance = -1;
812                    
813                    // looking for the farthests from main actor
814                    for(int i=1; i<this.getActorNumber(); i++){
815                            WPoint position_aux = this.getActorByIndex(i).getPosition();
816                            distance = Math.sqrt(
817                                            Math.pow(position_aux.getLatitudeE6() - getMainActor().getPosition().getLatitudeE6(), 2)
818                                            +
819                                            Math.pow(position_aux.getLongitudeE6() - getMainActor().getPosition().getLongitudeE6(), 2));
820                            
821                            // check the distance
822                            if ( distance > farthest_distance)
823                            {
824                                    farthest_distance = distance;
825                                    second_farthest = first_farthest;
826                                    first_farthest = i;
827                            }
828                    }
829                    
830                    // We return the value that user requests
831                    if(flag == FIRST_FARTHEST_ACTOR){
832                            return_value = first_farthest;
833                    }
834                    else if(flag == SECOND_FARTHEST_ACTOR){
835                            return_value = second_farthest;
836                    }
837                    return return_value;
838            }
