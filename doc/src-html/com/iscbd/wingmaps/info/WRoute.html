361            /**
362             * Set a new route color using ARGB
363             * @param alfa_p
364             * @param red_p
365             * @param green_p
366             * @param blue_p
367             */
368            public void setRouteColor(int alfa_p, int red_p, int green_p, int blue_p)
369            {
370                    int alfa = alfa_p;
371                    int red = red_p;
372                    int red2 = red;
373                    int red3 = red+100;
374                    int red4 = red;
375                    int green = green_p;
376                    int green2 = green;
377                    int green3 = green+100;
378                    int green4 = green+200;
379                    int blue = blue_p;
380                    int blue2 = blue;
381                    int blue3 = blue+100;
382                    int blue4 = blue+200;
383                    
384                    // red limit control
385                    if(red < 0) red = 0;
386                    else if(red > 255)   red = 255;
387                    if(red < 0) red = 0;
388                    else if(red2 > 255)  red2 = 255;
389                    if(red3 < 0) red3 = 0;
390                    else if(red3 > 255)  red3 = 255;
391                    if(red4 < 0) red4 = 0;
392                    else if(red4 > 255)  red4 = 255;
393                    
394                    // green limit control
395                    if(green < 0) green = 0;
396                    else if(green > 255) green = 255;
397                    if(green2 < 0) green2 = 0;
398                    else if(green2 > 255)        green2 = 255;
399                    if(green3 < 0) green3 = 0;
400                    else if(green3 > 255)        green3 = 255;
401                    if(green4 < 0) green4 = 0;
402                    else if(green4 > 255)        green4 = 255;
403                    
404                    // blue limit control
405                    if(blue < 0) blue = 0;
406                    else if(blue > 255)  blue = 255;
407                    if(blue2 < 0) blue2 = 0;
408                    else if(blue2 > 255) blue2 = 255;
409                    if(blue3 < 0) blue3 = 0;
410                    else if(blue3 > 255) blue3 = 255;
411                    if(blue4 < 0) blue4 = 0;
412                    else if(blue4 > 255) blue4 = 255;
413                    
414                    // set the brush
415                    p.setARGB(alfa, red, green, blue);
416                    p2.setARGB(alfa, red2, green2, blue2);
417                    p3.setARGB(255, red3, green3, blue3);
418                    p4.setARGB(255, red4, green4, blue4);
419            }

421            /**
422             * Build a http query to get a json result
423             * @param from_p
424             * @param to_p
425             * @param mode_p
426             * @return
427             */
428            private String buildQuery(WPoint from_p, WPoint to_p, String mode_p)
429            {
430                    String query = new String("https://maps.googleapis.com/maps/api/directions/json?origin=");
431            query = query + from.getLatitudeE6()/1E6 + "," + from.getLongitudeE6()/1E6;
432            query = query + "&destination=" + to.getLatitudeE6()/1E6;
433            query = query + "," + to.getLongitudeE6()/1E6;
434            query = query + "&sensor=false&mode="+mode_p;
435            return query;
436            }

438            /**
439             * Send the http query to get a json
440             * @param query
441             * @return
442             */
443            private String requestJSON(String query)
444            {
445                    StringBuilder builder = new StringBuilder();
446                    HttpClient client = new DefaultHttpClient();
447                    HttpGet httpGet = new HttpGet(query);
448                    try {
449                            HttpResponse response = client.execute(httpGet);
450                            StatusLine statusLine = response.getStatusLine();
451                            int statusCode = statusLine.getStatusCode();
452                            if (statusCode == 200) {
453                                    HttpEntity entity = response.getEntity();
454                                    InputStream content = entity.getContent();
455                                    BufferedReader reader = new BufferedReader(
456                                                    new InputStreamReader(content));
457                                    String line;
458                                    while ((line = reader.readLine()) != null) {
459                                            builder.append(line);
460                                    }
461                            } else {
462                                    System.out.println("Failed to download json-file");
463                            }
464                    } catch (ClientProtocolException e) {
465                            e.printStackTrace();
466                    } catch (IOException e) {
467                            e.printStackTrace();
468                    }
469                    return builder.toString();
470            }

473            /**
474             * Decode a json string into a json object
475             * @param json_p
476             * @throws JSONException
477             */
478            private void readJSON(String json_p) throws JSONException{
479                    json = new JSONObject(json_p);
480                    
481                    JSONObject better_route = json.getJSONArray("routes").getJSONObject(0);
482                    JSONArray steps_aux = better_route.getJSONArray("legs").getJSONObject(0).getJSONArray("steps");
483                    
484                    distance = better_route.getJSONArray("legs").getJSONObject(0).getJSONObject("distance").getString("text");
485                    duration = better_route.getJSONArray("legs").getJSONObject(0).getJSONObject("duration").getString("text");
486                    Double start_lat = new Double(better_route.getJSONArray("legs").getJSONObject(0).getJSONObject("start_location").getString("lat"));
487                    Double start_lon = new Double(better_route.getJSONArray("legs").getJSONObject(0).getJSONObject("start_location").getString("lng"));
488                    Double end_lat = new Double(better_route.getJSONArray("legs").getJSONObject(0).getJSONObject("end_location").getString("lat"));
489                    Double end_lon = new Double(better_route.getJSONArray("legs").getJSONObject(0).getJSONObject("end_location").getString("lng"));
490                    from = new WPoint((int)(1E6*start_lat.doubleValue()),(int)(1E6*start_lon.doubleValue()));
491                    to = new WPoint((int)(1E6*end_lat.doubleValue()),(int)(1E6*end_lon.doubleValue()));
492                    
493                    num_steps = steps_aux.length();
494                    
495                    // tomamos la informacion de cada paso
496                    for(int i=0; i<num_steps; i++){
497                            // localizamos el paso actual dentro del JSON
498                            JSONObject actual_step = steps_aux.getJSONObject(i);
499                            
500                            Double start_lat_aux = new Double(actual_step.getJSONObject("start_location").getString("lat"));
501                            Double start_lon_aux = new Double(actual_step.getJSONObject("start_location").getString("lng"));
502                            Double end_lat_aux = new Double(actual_step.getJSONObject("end_location").getString("lat"));
503                            Double end_lon_aux = new Double(actual_step.getJSONObject("end_location").getString("lng"));
504                            
505                            WPoint start_paso = new WPoint((int)(1E6*start_lat_aux.doubleValue()),(int)(1E6*start_lon_aux.doubleValue()));
506                            WPoint end_paso = new WPoint((int)(1E6*end_lat_aux.doubleValue()),(int)(1E6*end_lon_aux.doubleValue()));
507                            
508                            String instructions_aux = new String(actual_step.getString("html_instructions"));
509                            String travelMode_aux = new String(actual_step.getString("travel_mode")); 
510                            
511                            String polyline = new String(actual_step.getJSONObject("polyline").getString("points"));
512                            
513                            WRouteSteps aux_step = new WRouteSteps(actual_step.getJSONObject("distance").getString("text"),
514                                            actual_step.getJSONObject("duration").getString("text"),start_paso, end_paso, instructions_aux, travelMode_aux, polyline);
515                            
516                            // steps = new ;
517                            steps.add(aux_step);
518                    }
519                    
520            }

600            /**
601             * Draws the route properly
602             * @param canvas
603             * @param map_view
604             * @param shadow
605             * @param when
606             */
607            public void draw(Canvas canvas, MapView map_view, boolean shadow, long when)
608            {
609                    int anchura, anchura2, anchura3;
610                    
611                    route_width = new Double(100*((4*591657550.5/1E6)/2));
612                    Integer prop = new Integer((int)Math.pow((double)2,(double)map_view.getZoomLevel()-1));
613                    route_width = route_width / prop;
614                    
615                    Projection projection = map_view.getProjection();
616                    
617                    if(map_view.getZoomLevel() < 15)
618                    {
619                            anchura = 50;
620                            anchura2 = 40;
621                            anchura3 = 4;
622                    }
623                    else
624                    {
625                            anchura = 25;
626                            anchura2 = 20;
627                            anchura3 = 2;
628                    }
629                    
630                    // asignamos la anchura dependiendo del zoom del mapa
631                    //p.setStrokeWidth(anchura/route_width.floatValue());
632                    p.setStrokeWidth(10);
633                    p.setStyle(Paint.Style.STROKE);
634                    
635                    Path path_route = new Path();
636                    
637                    Point aux_point = new Point();
638                    projection.toPixels(steps.get(0).getFrom(), aux_point);
639                    path_route.moveTo(aux_point.x, aux_point.y);
640                    projection.toPixels(steps.get(0).getTo(), aux_point);
641                    path_route.lineTo(aux_point.x, aux_point.y);
642                    
643                    for(int i=1; i<this.getNumSteps(); i++){
644                            aux_point = new Point();
645                            
646                            projection.toPixels(steps.get(i).getFrom(), aux_point);
647                            path_route.lineTo(aux_point.x, aux_point.y);
648                            
649                            for(WPoint aux_w : steps.get(i).polyline)
650                            {
651                                    projection.toPixels(aux_w, aux_point);
652                                    path_route.lineTo(aux_point.x, aux_point.y);
653                            }
654                            projection.toPixels(steps.get(i).getTo(), aux_point);
655                            path_route.lineTo(aux_point.x, aux_point.y);
656                    }
657                    
658                    
659                    path_route.setFillType(Path.FillType.WINDING);
660                    canvas.drawPath(path_route, p);
661                    
662            }
