148                    {
149                            for(int i=0; i< fields.size(); i++){

234                                            }
235                                            else{
236                                                    lines.add(aux_word);

238                                    }
239                                    else if((word_width + line_width + space_width) > width){
240                                            line_index++;
241                                            lines.add(aux_word);
242                                    }
243                                    else if((word_width) > width){
244                                            System.out.println("********************************");
245                                            width = word_width;
246                                    }
247                                    else{
248                                            lines.add(aux_word);
249                                            line_index++;
250                                    }

358                    /**

360                     * @param parent_p
361                     * @param icon_p
362                     * @param position

365                    WButton(WBubble parent_p, Bitmap icon_p, WPoint position, Intent intent_p)
366                    {
367                            parent = parent_p;

371                            intent = intent_p;

374                    //  ***   M E T H O D S
375                    private void calculateXY(MapView mapView, WPoint position_p, int index_p)
376                    {
377                            Projection projection = mapView.getProjection();
378                            Point point = new Point();
379                            projection.toPixels(position_p, point);
380                            
381                            x = point.x;
382                            y = point.y;
383                            
384                            if(isActorBubble())
385                                    y = y - arrow_height - margin_bottom - actor.getIcon().getHeight() - button_icon.getHeight();
386                            else if(isPlaceBubble())
387                                    y = y - arrow_height - margin_bottom - place.getIcon().getHeight() - button_icon.getHeight();
388                                    
389                            x = x + interior_width/2 - margin_right - button_icon.getWidth();
390                            for(int i = 0; i < buttons.size(); i++)
391                            {
392                                    if(i != index_p)
393                                            x = x - parent.buttons.get(i).getBitmap().getWidth() - margin_buttons;
394                                    if(i >= index_p)
395                                            break;
396                            }
397                    }
398                    
399                    public boolean isHit(WPoint position_p, MapView mapView)
400                    {
401                            Projection projection = mapView.getProjection();
402                            Point hitpoint = new Point();
403                            projection.toPixels(position_p, hitpoint);
404                            
405                            if(hitpoint.x > (x - hitmargin) && hitpoint.x < (x+button_icon.getWidth() + hitmargin) 
406                                            && hitpoint.y > (y - hitmargin) && hitpoint.y < (y+button_icon.getHeight() + hitmargin))
407                            {
408                                    return true;
409                            }
410                            else
411                                    return false;
412                    }
413                    
414                    public float getX()
415                    {
416                            return x;
417                    }
418                    
419                    public float getY()
420                    {
421                            return y;
422                    }
423                    
424                    public Intent getIntent()
425                    {
426                            return intent;
427                    }
428                    
429                    public void setIcon(Bitmap icon_p)
430                    {
431                            Bitmap final_icon;
432                                    
433                            // resize image
434                            double icon_prop = ((float) icon_p.getWidth()) / ((float)icon_p.getHeight());
435                            if(icon_p.getHeight() >= icon_p.getWidth())
436                            {
437                                    if(icon_p.getHeight() >= MAX_BUTTON_HEIGHT)
438                                    {
439                                            double aux_width = MAX_BUTTON_HEIGHT * icon_prop;
440                                            final_icon = Bitmap.createScaledBitmap(icon_p, (int) aux_width, MAX_BUTTON_HEIGHT, false);
441                                    }
442                                    else
443                                            final_icon = icon_p;
444                            }
445                            else
446                            {
447                                    if(icon_p.getWidth() >= MAX_BUTTON_WIDTH)
448                                    {
449                                            double aux_height = (MAX_BUTTON_WIDTH / icon_prop);
450                                            final_icon = Bitmap.createScaledBitmap(icon_p, MAX_BUTTON_WIDTH, (int) aux_height, false);
451                                    }
452                                    else
453                                            final_icon = icon_p;
454                            }
455                                    
456                                            
457                            button_icon = final_icon;
458                    }
459                    
460                    public Bitmap getBitmap()
461                    {
462                            return button_icon;
463                    }
464                    
465                    public int getWidth()
466                    {
467                            return button_icon.getWidth();
468                    }
469                    
470                    public int getHeight()
471                    {
472                            return button_icon.getHeight();
473                    }
